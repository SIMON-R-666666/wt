        <label for="separator">Choose a separator:</label>
        <select id="separator" name="separator" required>
            <option value="#">#</option>
            <option value="|">|</option>
            <option value="%">%</option>
        </select><br><br>

        <label for="newSeparator">Enter a new separator for replacement:</label>
        <input type="text" id="newSeparator" name="newSeparator" maxlength="1" required><br><br>

<?php
if ($_SERVER["REQUEST_METHOD"] == "POST") {
    // Collect data from form
    $inputString = $_POST['inputString'];
    $separator = $_POST['separator'];
    $newSeparator = $_POST['newSeparator'];

    // a. Split the string into separate words using the given separator
    $wordsArray = explode($separator, $inputString);
    echo "The string split into words:<br>";
    foreach ($wordsArray as $word) {
        echo $word . "<br>";
    }

    // b. Replace all occurrences of the separator in the string with a new separator
    $replacedString = str_replace($separator, $newSeparator, $inputString);
    echo "String after replacing '$separator' with '$newSeparator':<br>";
    echo $replacedString . "<br>";

    // c. Find the last word in the given string
    $lastWord = end($wordsArray);
    echo "The last word in the string is: $lastWord";
}
?>
