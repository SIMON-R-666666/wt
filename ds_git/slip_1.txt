import pandas as pd
import matplotlib.pyplot as plt
# Load the Iris dataset from the CSV file
data = pd.read_csv(r'C:\Users\User\Desktop\vivek\ds\iris.csv')
# Calculate the frequency of each species
sp_cnt= data['species'].value_counts()
# Create a pie chart
plt.figure(figsize=(6, 6))
plt.pie(sp_cnt, labels=sp_cnt.index, autopct='%1.1f%%', startangle=140)
plt.title('Iris Species Frequency')
# Display the pie chart
plt.show()
